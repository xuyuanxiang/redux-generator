/**
 * @module redux-generator
 * @description
 * @version 1.0.0
 * @author xuyuanxiang
 * @date 16/10/8
 */
import isFunction from 'lodash/isFunction';
import isEqual from 'lodash/isEqual';
import toString from 'lodash/toString';
import isPlainObject from 'lodash/isPlainObject';
 
export default store => next => action => {
    if (!isFunction(action)) {
        return next(action);
    }
    const generator = action(store.dispatch, store.getState);
    if (!generator || !isFunction(generator.next)) {
        return next(action);
    }
    let done = false, limits = 0, value;
    while (done === false) {
        let tmp = generator.next();
        if (tmp && typeof value !== 'undefined' && isEqual(tmp.value, value)) {
            if (++limits === 3) {
                const error = new Error(`The generator.next() return the same value over ${limits} times: ${isPlainObject(value) ? JSON.stringify(value) : toString(value)} `);
                console.error(error);
                throw error;
            }
        }
        value = tmp ? tmp.value : undefined;
        done = !tmp || tmp.done;
        if (typeof value === 'undefined') continue;
        next(value);
    }
}
